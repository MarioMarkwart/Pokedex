async function renderFoundPokemon(pokemonToSearch){
    document.getElementById('overview-container').innerHTML = "";
    let foundPokemon = [];
    for(let i=0; i < pokemonToSearch.length; i++) {
        let url = pokemonToSearch[i];
        let response = await fetch(url);
        let responseAsJson = await response.json();
        foundPokemon.push(responseAsJson);
        setPokemonInformations(responseAsJson);
    }

    // rendern
}
